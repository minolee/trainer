    model_config = ConfigDict(extra="allow")

    speaker: str

    @field_validator("speaker")
    def process_speaker(cls, v):
        v = v.lower()
        if v == "midm":
            v = "assistant"
        return v



    def __str__(self):
        return f"{self.speaker}: {self.message}"
    
    def __repr__(self):
        return str(self)
